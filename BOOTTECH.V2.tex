import qutip as qt
import numpy as np
import matplotlib.pyplot as plt

N = 18  # multi-knot lattice (trefoil + borromean)
J, h, theta = 1.0, 0.03, 6*np.pi/5

# Multi-knot links
links = [(i,(i+1)%N) for i in range(N)] + [(0,9),(3,12),(6,15)]  # trefoil + borromean crossings

H = qt.tensor([qt.qzero(2)] * N)
for i,j in links:
    op = qt.tensor([qt.qeye(2)] * N)
    op = qt.tensor_replace(op, [qt.sigmaz() if k in [i,j] else qt.qeye(2) for k in range(N)])
    H += -J * op

for i in range(N):
    op = qt.tensor([qt.qeye(2)] * N)
    op = qt.tensor_replace(op, [qt.sigmax() if k==i else qt.qeye(2) for k in range(N)])
    H += h * op

psi0 = qt.tensor([(qt.basis(2,0) + qt.basis(2,1)).unit() for _ in range(N)])

times = np.linspace(0, 100, 2000)
result = qt.mesolve(H, psi0, times)

# Pulsed emission proxy
pulsed = [abs(state.overlap(qt.tensor([qt.sigmax() if k%5==0 else qt.qeye(2) for k in range(N)])))**2 for state in result.states]

plt.figure(figsize=(12,7))
plt.plot(times, pulsed, label='Pulsed Intensity (Lk=100\% saturation)')
plt.title('Indestructible Topological Pulsar – Perfect Pulsation')
plt.xlabel('Time')
plt.ylabel('Intensity')
plt.legend()
plt.grid(True)
plt.show()

# Final entanglement entropy
rho = result.states[-1]
S = qt.entropy_vn(rho.ptrace(list(range(N//2))))
print(f"Saturation entanglement entropy ≈ {S:.4f} (target ln4 ≈ 1.3863)")
